  const [data, setData] = useRecoilState(dataState);

  return <div>{data}</div>;
};

  const [data, setData] = useRecoilState(dataState);
  const [someDependency] = useRecoilState(someDependencyState);

      setData(someDependency);

  }, [someDependency]);

  return <></>;
};
